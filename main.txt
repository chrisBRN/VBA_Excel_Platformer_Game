Option Explicit

' /////////////////////////////////////////////////////////////////////////////
' GLOBALS /////////////////////////////////////////////////////////////////////
' /////////////////////////////////////////////////////////////////////////////

Public Const SHEET_NAME = "Important Work"

Private Const PLAYER_COLOUR = rgbDimGrey
Private Const GROUND_COLOUR = rgbDarkGray

' /////////////////////////////////////////////////////////////////////////////
' PROGRAM /////////////////////////////////////////////////////////////////////
' /////////////////////////////////////////////////////////////////////////////

Sub main()

    Dim window As clsWindow
    Set window = New clsWindow
        
  
    
    Dim ground As Range
    
    Dim bLCell As String
    Dim bRCell As String
    
    bLCell = window.get_bottom_left_cell
    bRCell = window.get_bottom_right_cell
    
    Dim groundThickness As Integer
    
    groundThickness = window.get_cells_high / 5
    
    With Worksheets(SHEET_NAME)
        Set ground = Range(bLCell, bRCell).Offset(-groundThickness, 0).Resize(groundThickness)
    End With
    
    ground.Interior.Color = GROUND_COLOUR
    
    
End Sub
